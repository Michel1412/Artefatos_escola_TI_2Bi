actor "SDK (Aplicação Cliente)" as sdk

rectangle "LogViewer SaaS Core" {
    
    ' Grupo: Gestão de Acesso e Assinatura
    usecase "Gerir Usuários" as UC_User
    usecase "Autenticar c/ 2FA e Aprovação Mobile" as UC_Auth
    usecase "Gerenciar Assinatura e Planos" as UC_Billing
    
    ' Grupo: Contexto e Visualização
    usecase "Configurar e Gerenciar Contexto" as UC_Context
    usecase "Visualizar Dashboard (Web/Mobile)" as UC_Dash
    usecase "Gerar e Exportar Relatórios (PDF/Excel)" as UC_Report
    
    ' Grupo: Interação com Logs e Erros
    usecase "Explorar e Filtrar Histórico de Erros" as UC_Explore
    usecase "Gerenciar Status, Leitura e Arquivamento" as UC_ManageErrors
    usecase "Interagir com IA para Sugestão de Solução" as UC_IA_Interact
    
    ' Grupo: Captura e Notificação (SDK / Sistema)
    usecase "Integrar e Capturar Logs (Em Lote/Streaming)" as UC_Capture
    usecase "Notificar Erros Encontrados (Push/Email)" as UC_Notify
    
    ' Grupo: Processamento de IA
    usecase "Identificar Logs Sem Padrões Estruturados" as UC_Identify
    usecase "Processar Contexto, Sanitizar Senhas e Agrupar" as UC_Process
}

' --- RELACIONAMENTOS ---

' Ações diretas do Usuário
user --> UC_User
user --> UC_Auth
user --> UC_Billing
user --> UC_Context
user --> UC_Dash
user --> UC_Report
user --> UC_Explore
user --> UC_ManageErrors
user --> UC_IA_Interact

' Ações da SDK
sdk --> UC_Capture
sdk --> UC_Notify

' Ações do Agente IA
ia --> UC_Identify
ia --> UC_Process
UC_IA_Interact <-- ia : Responde com Solução

' Gateway de Pagamento
UC_Billing --> pagto : Processa Pagamento
