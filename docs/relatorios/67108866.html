Table Empresas {
  id_empresa uuid [pk, default: `gen_random_uuid()`]
  cnpj varchar(14) [not null, unique] 
  razao_social varchar(255) [not null]
  id_plano int [not null]
  data_cadastro timestamptz [not null, default: `now()`]
}

Table Sistemas_Cliente {
  id_sistema uuid [pk, default: `gen_random_uuid()`]
  id_empresa uuid [not null]
  nome_sistema varchar(100) [not null]
  caminho_pasta_logs varchar(255) [not null] 
}

Table Erros_Armazenados {
  id_erro uuid [pk, default: `gen_random_uuid()`]
  id_sistema uuid [not null]
  id_log_mongo varchar(24) [not null, unique] 
  tipo_erro varchar(155) [not null] 
  conteudo_log_ofuscado text [not null] 
  contexto_aplicado_snapshot text [not null] 
  criticidade CriticidadeLog [not null, default: `CriticidadeLog.BAIXO`] 
  status StatusErro [not null, default: `StatusErro.NAO_SOLUCIONADO`] 
  contador_repeticoes int [not null, default: 1] 
  link_agente_llm varchar(512)
  sugestao_solucao text 
  data_ocorrencia timestamptz [not null, default: `now()`] 
  
  Note: 'Gerido por Serviço de Erro (Peng) e Serviço de tratamento do log (pedroalvaro193)'
}

  email varchar(255) [not null, unique]
  perfil_role Role [not null, default: `Role.DEV`] 
  dois_fatores_ativo boolean [not null, default: false] 
  
  Note: 'Mapeado na classe User (pedroalvaro193) & Serviço de usuário (Michel)'

  id_solicitacao uuid [pk, default: `gen_random_uuid()`]

  data_criacao timestamptz [not null, default: `now()`]
  data_expiracao timestamptz [not null] 
  ip_origem varchar(45) [not null] 
  
  Note: 'Controlado por Autenticação de 2 fatores (Michel Bocchi Junior)'
}

Table Tokens_SDK {
  id_token uuid [pk, default: `gen_random_uuid()`]
  id_sistema uuid [not null, unique]
  token_hash varchar(255) [not null, unique] 
  ativo boolean [not null, default: true]
  data_geracao timestamptz [not null, default: `now()`]
  
  Note: 'Alimentado pela classe Serviço User Token (Peng)'
}

Table Contextos_IA {
